CREATE TABLE USUARIO (
    ID_USUARIO INTEGER       PRIMARY KEY AUTOINCREMENT,
    NOME       VARCHAR (100),
    EMAIL      VARCHAR (100),
    SENHA      VARCHAR (100) 
);



CREATE TABLE REMESSA (
    ID_REMESSA       INTEGER         PRIMARY KEY AUTOINCREMENT,
    DESCRICAO        VARCHAR (100),
    ORIGEM           VARCHAR (100),
    DESTINO          VARCHAR (100),
    VALOR            DECIMAL (9, 2),
    STATUS           CHAR (1),
    ORIGEM_LATITUDE  DECIMAL (12, 8),
    ORIGEM_LONGITUDE DECIMAL (12, 8),
    ID_USUARIO       INTEGER,
    DT_CADASTRO      DATETIME,
    ID_ENTREGADOR    INTEGER
);
